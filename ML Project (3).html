def outliertreat(df,col): 
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1
    UL = Q3 + IQR
    LL = Q1 - IQR
    df.loc[df[col]>UL,col] = df[col].median()
    df.loc[df[col]<LL,col] = df[col].median()
